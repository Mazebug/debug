#include <stdio.h>
#include <string.h>

int main()
{
    int n;
    scanf("%d", &n);
    int i, j, k;
    char a[100][100]={'0'};
    char temp[100] = {'0'};

    i = 0;
    j = 0;
    while (scanf("%c", &a[i][j]) != EOF) {
        for (i = 0; i < n; i++) {
            for (j = 0; j < 101; j++) {
                scanf("%c", &a[i][j]);
                if (a[i][j] == '\n') {
                    a[i][j] = '\0';
                    break;
                }
            }
        }
    }
    
    for (i = 0, k = 0; i < n - 1; i++) {
        if (strlen(a[k]) > strlen(a[i + 1])) {
            strcpy(temp, a[k]);
        } else {
            k = i + 1;
            strcpy(temp, a[k]);
        }
    }
    printf("%s", temp);

    return 0;
}
