#include <stdio.h>
#include <string.h>

    int main() 
    {
        int n;
        scanf("%d", &n);
        char names[n][100];
        char longest[100];
        int z = 0;
        int max = 0;
        int i = 0;
        while (scanf("%c", &names[i][z]) != EOF) {
            while (names[i][z] != '\n') {
                    z++;
                    scanf("%c", &names[i][z]);
            }
            if(z > max) {
                max = z;
                strcpy(longest, names[i]);
            }
            i++;
            z = 0;
        }  
        for(int j = 0; j < strlen(longest); j++)
            printf("%c", longest[j]);
        return 0;
    }
